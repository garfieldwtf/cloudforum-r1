Deploying to Cloudflare with R1 Database

This forum is designed to work with Cloudflare Pages and R1 SQL database. Follow these steps:

  1. Create a GitHub repository and upload your forum files
  2. Log in to your Cloudflare dashboard and select "Pages"
  3. Click "Create a project" and connect your GitHub account
  4. Select the repository containing your forum
  5. Configure build settings (if needed) and click "Deploy"
  6. Create an R1 database in the Cloudflare dashboard
  7. Create a Worker to handle database operations

Example Worker Code for R1 Database

// worker.js - Cloudflare Worker for R1 SQL database
export default {
async fetch(request, env) {
const url = new URL(request.url);

// Handle CORS preflight requests
if (request.method === 'OPTIONS') {
return new Response(null, {
headers: {
'Access-Control-Allow-Origin': '*',
'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
'Access-Control-Allow-Headers': 'Content-Type',
},
});
}

// Handle API routes
if (url.pathname === '/api/threads' && request.method === 'GET') {
// Get all threads from R1 database
try {
const result = await env.DB.prepare('SELECT * FROM threads ORDER BY created_at DESC').all();
return new Response(JSON.stringify(result), {
headers: {
'Content-Type': 'application/json',
'Access-Control-Allow-Origin': '*'
}
});
} catch (error) {
return new Response(JSON.stringify({ error: error.message }), {
status: 500,
headers: {
'Content-Type': 'application/json',
'Access-Control-Allow-Origin': '*'
}
});
}
}

if (url.pathname === '/api/threads' && request.method === 'POST') {
// Create a new thread in R1 database
try {
const data = await request.json();
const { title, content, author } = data;

const result = await env.DB.prepare(
'INSERT INTO threads (title, content, author) VALUES (?, ?, ?)'
).bind(title, content, author).run();

return new Response(JSON.stringify({ success: true, id: result.meta.last_row_id }), {
headers: {
'Content-Type': 'application/json',
'Access-Control-Allow-Origin': '*'
}
});
} catch (error) {
return new Response(JSON.stringify({ error: error.message }), {
status: 500,
headers: {
'Content-Type': 'application/json',
'Access-Control-Allow-Origin': '*'
}
});
}
}

return new Response('Not found', { status: 404 });
}
}

Database Schema

-- SQL to create the necessary tables in R1
CREATE TABLE threads (
id INTEGER PRIMARY KEY AUTOINCREMENT,
title TEXT NOT NULL,
content TEXT NOT NULL,
author TEXT NOT NULL,
created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE posts (
id INTEGER PRIMARY KEY AUTOINCREMENT,
thread_id INTEGER NOT NULL,
content TEXT NOT NULL,
author TEXT NOT NULL,
created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
FOREIGN KEY (thread_id) REFERENCES threads (id)
);

CREATE TABLE users (
id INTEGER PRIMARY KEY AUTOINCREMENT,
username TEXT UNIQUE NOT NULL,
email TEXT UNIQUE NOT NULL,
created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
